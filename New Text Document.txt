   @Override
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == proceedButton) {
            // Proceed button clicked
            if (canProceed) {
                index++;
                nextQuestion();
                canProceed = false; // Reset the flag after proceeding to the next question
            } else {
                // Show error message if the user tries to proceed without selecting two correct answers
                JOptionPane.showMessageDialog(frame, "Please select at least two correct answers before proceeding.");
            }
        } else {
            // Checkbox clicked
            JCheckBox selectedCheckbox = (JCheckBox) e.getSource();
            char selectedAnswer = ' ';
            if (selectedCheckbox == checkboxA) {
                selectedAnswer = '1';
            } else if (selectedCheckbox == checkboxB) {
                selectedAnswer = '2';
            } else if (selectedCheckbox == checkboxC) {
                selectedAnswer = '3';
            } else if (selectedCheckbox == checkboxD) {
                selectedAnswer = '4';
            }

            // Toggle the canProceed flag if the selected checkbox has a correct answer
            boolean correct = false;
            for (char correctAnswer : Atbildes[index]) {
                if (selectedAnswer == correctAnswer) {
                    correct = true;
                    break;
                }
            }

            if (!correct) {
                // Wrong answer, display retry message
                JOptionPane.showMessageDialog(frame, "Incorrect answer. Please retry this question.");
            } else {
                // Check how many checkboxes have correct answers selected
                int correctCount = 0;
                if (checkboxA.isSelected()) correctCount++;
                if (checkboxB.isSelected()) correctCount++;
                if (checkboxC.isSelected()) correctCount++;
                if (checkboxD.isSelected()) correctCount++;

                // Enable the "Proceed" button if at least two checkboxes have correct answers selected
                canProceed = (correctCount >= 2);
                proceedButton.setEnabled(canProceed);
            }
        }
    }
